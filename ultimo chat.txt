Supabase SQL Schema (ejecutar en Dashboard)
sql
CREATE TABLE active_patterns (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  heart_rate NUMERIC,
  breath_rate NUMERIC,
  variability NUMERIC,
  location JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  expires_at TIMESTAMPTZ DEFAULT NOW() + INTERVAL '5 minutes'
);
CREATE INDEX idx_active_patterns_expires ON active_patterns(expires_at);
CREATE INDEX idx_active_patterns_user ON active_patterns(user_id);

Supabase SQL Schema
sql
CREATE TABLE sync_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_code VARCHAR(6) UNIQUE NOT NULL,
  user1_id UUID REFERENCES auth.users(id),
  user2_id UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  started_at TIMESTAMPTZ,
  ended_at TIMESTAMPTZ,
  duration_seconds INTEGER,
  avg_sync_percentage NUMERIC,
  sync_data JSONB
);
CREATE TABLE sync_realtime (
  session_id UUID REFERENCES sync_sessions(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id),
  timestamp TIMESTAMPTZ DEFAULT NOW(),
  breath_rate NUMERIC,
  heart_rate NUMERIC,
  PRIMARY KEY (session_id, user_id, timestamp)
);
